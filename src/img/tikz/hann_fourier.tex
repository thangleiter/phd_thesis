\begin{tikzpicture}
    \tikzset{font = \footnotesize}
    \begin{axis} [
            declare function={
                T = 1;
                hann(\x) = 4 * pi^2 / \x * sin(deg(\x * T / 2)) / (4 * pi^2 - \x^2 * T^2);
            },
            xlabel={$\omega_n$},
            ylabel={$\hat{w}_n$},
            x label style={at={(current axis.south east)},anchor = north, xshift=2pt},
            y label style={at={(current axis.north west)},anchor = south, yshift=2pt},
            axis lines=middle,
            %line width = 1 pt,
            %scale only axis, % Ensures the axis region is exactly \marginparwidth wide
            width=1.15\marginparwidth,
            height=5cm,
            domain=-11*pi:11*pi,
            samples=200,
            xmin=-12*pi, xmax=12*pi,
            ymin=-0.25/2, ymax=1.25/2,
            xtick={-10*pi, 0, 10*pi},
            xticklabels={$\flatfrac{-10\pi}{T}$, $0$, $\flatfrac{10\pi}{T}$},
            ytick={0, 0.5},
            yticklabels={$$, $\flatfrac{T}{2}$},
            every tick/.style={
                black,
                thick
            },
        ]

        \addplot [RWTHmagenta100, very thick] {hann(x)};

        \foreach \X in {1,2} {
            \pgfmathsetmacro{\y}{hann(\X)}
            \addplot[mark = *, mark options = {RWTHgreen50}] coordinates {(\X,\y)};
        }

    \end{axis}
